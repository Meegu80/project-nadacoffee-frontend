MenuPage.tsx (NEW) 
// [판단 기준] 상품 등록일로부터 30일 이내 여부 계산
const isNewProduct = (createdAt: string) => {
  const oneMonthAgo = new Date();
  oneMonthAgo.setMonth(oneMonthAgo.getMonth() - 1);
  return new Date(createdAt) >= oneMonthAgo;
};

const isNew = isNewProduct(product.createdAt) && !isSoldOut;

// [UI 구현] Layered Star & Text Rotation
{isNew && (
  <div className="absolute bottom-6 right-6 rotate-12">
    <Star size={96} className="fill-brand-yellow" />
    <span className="absolute inset-0 flex items-center 
      justify-center font-black">NEW</span>
  </div>
)}
동적 기간 산출 로직
서버의 `createdAt` 데이터를 현재 시점과 비교하여, 매일 자정 자동으로 배지가 활성/비활성되는 자동화 로직 구축.
기하학적 레이어 위계
Star SVG 아이콘 위에 텍스트를 중첩하고 레이어 전체에 12도 회전을 주어, 평면적인 카드 UI에 입체감과 생동감 부여.
별 모양 아이콘의 채움색(`fill`)을 브랜드 컬러로 지정하여 통일감 있는 마케팅 메시지 전달.
