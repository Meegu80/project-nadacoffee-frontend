MenuPage.tsx (Logic) 
// 1. 디바운싱: 500ms 동안 입력이 없으면 검색 실행
useEffect(() => {
  const handler = setTimeout(() => { 
    setDebouncedSearchQuery(searchQuery); 
  }, 500);
  return () => clearTimeout(handler);
}, [searchQuery]);

// 2. TanStack Query 기반 비동기 데이터 조회
const { data: searchAutocompleteData } = useQuery({
  queryKey: ['products', 'autocomplete', debouncedSearchQuery],
  queryFn: () => getProducts({ 
    isDisplay: 'true', limit: 10, search: debouncedSearchQuery 
  }),
  enabled: !!debouncedSearchQuery.trim(),
  staleTime: 1000 * 60 // 1분 캐싱
});
유예 시간 제어 (Debouncing)
입력마다 API를 호출하지 않고, 사용자의 타이핑 완료 시점을 예측하여 네트워크 자원을 효율적으로 관리하는 필수 전략 적용.
지능형 데이터 캐싱
React Query의 staleTime을 활용하여 동일한 키워드 재입력 시 API 호출 없이 메모리에서 즉각 응답하여 UX 체감 속도 향상.
불필요한 리렌더링과 데이터 통신을 방지하여 프론트엔드와 백엔드 양측의 부하를 동시에 최적화함.
