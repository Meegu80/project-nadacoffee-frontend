// [핵심 UX 로직] 헤더 클릭 시 즉각적인 데이터 정렬/필터링 연동
<th 
  className="px-6 py-4 cursor-pointer hover:bg-gray-100 transition-all group" 
  onClick={() => handleOrderSort('amount')}
>
  <div className="flex items-center justify-center group-hover:text-[#222222]">
    결제금액 
    // 활성화된 필드에만 시각적 힌트(아이콘) 노출
    <OrderSortIcon field="amount" />
  </div>
</th>

// 데이터 정렬 정합성 보장 로직
const handleOrderSort = (field) => {
  if (orderSortField === field) 
    setOrderSortOrder(orderSortOrder === "asc" ? "desc" : "asc");
  else { 
    setOrderSortField(field); 
    setOrderSortOrder("asc"); 
  }
};
