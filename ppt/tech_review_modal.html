ReviewModal.tsx (Sequential Upload) 
// 1. 안정적인 순차적 이미지 업로드 로직
for (let i = 0; i < fileArray.length; i++) {
  const file = fileArray[i];
  try {
    // 병렬 과부하 방지를 위한 순차 처리
    const uploadedUrl = await uploadImage(file, 'reviews');
    
    setImages(prev => prev.map(img => 
      img.url === tempUrl ? { url: uploadedUrl, isUploading: false } : img
    ));
  } catch (error) {
    // 실패 시 에러 핸들링 및 상태 복구
    setImages(prev => prev.filter(img => img.url !== tempUrl));
  }
}

Sequential File Management
최대 5장의 이미지를 지원하며, 대용량 파일 업로드 시 발생할 수 있는 서버 과부하를 방지하기 위해 `for...of`를 활용한 순차적 비동기 업로드 메커니즘 구축.
Dynamic Rating Interaction
