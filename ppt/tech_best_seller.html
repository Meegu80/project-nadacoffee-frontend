// 1. 무한 루프를 위한 데이터 복제 및 분할 로직
const row1 = useMemo(() => {
  const items = finalDisplayItems.slice(0, 5);
  return Array(4).fill(items).flat(); // 4배 복제로 공백 방지
}, [finalDisplayItems]);

// 2. Framer Motion을 활용한 2축 마키(Marquee) 애니메이션
<motion.div
  className="flex gap-8 w-max"
  animate={{ x: ["-50%", "0%"] }} // 1행: 좌에서 우로 무한 이동
  transition={{ 
    repeat: Infinity, 
    duration: 60, 
    ease: "linear" 
  }}
>
  {row1.map((p, idx) => <ProductCard key={idx} product={p} />)}
</motion.div>

// 3. 양방향 교차 모션 설계 (2행: 우 -> 좌)
animate={{ x: ["0%", "-50%"] }}
Seamless Loop 설계
데이터를 4배수로 복제하고 CSS X축 위치를 -50%에서 0%로 반복 제어하여, 사용자가 끊김을 느끼지 못하는 완벽한 무한 루프 구현.
교차 방향 모션 (Bi-directional)
상단 행(Left-to-Right)과 하단 행(Right-to-Left)의 이동 방향을 반대로 설계하여 화면 전체에 입체감과 역동성을 부여.
랭킹 데이터 실시간 매핑
주문 통계 API에서 산출된 랭킹 데이터와 상품 상세 메타데이터를 클라이언트 사이드에서 실시간으로 결합 및 최신순 정렬.
