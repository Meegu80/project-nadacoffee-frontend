// 1. 최근 7일간의 매출 데이터 산출 로직
const chartData = useMemo(() => {
  if (!ordersData) return [];
  const today = new Date();
  const data = [];

  for (let i = 6; i >= 0; i--) {
    const d = new Date(today);
    d.setDate(today.getDate() - i);
    const dateStr = getKSTDateString(d);

    // 유효한 주문 상태 필터링 및 일자별 합계 계산
    const dailyTotal = ordersData.data
      .filter(o => getKSTDateString(o.createdAt) === dateStr &&
        ['PAYMENT_COMPLETED', 'PURCHASE_COMPLETED'].includes(o.status))
      .reduce((sum, o) => sum + (o.totalPrice || 0), 0);

    data.push({
      name: `${d.getMonth() + 1}/${d.getDate()}`,
      sales: dailyTotal
    });
  }
  return data;
}, [ordersData]);
슬라이딩 윈도우 생성
`for` 루프를 역순으로 수행하여 오늘을 기준으로 정확히 최근 7일치(D-6 ~ Today)의 날짜 배열을 생성함.
유효 매출 필터링
단순 주문이 아닌 결제완료, 배송완료, 구매확정 등 '실제 매출'로 집계되는 상태 값들만 선별하여 데이터 신뢰성 확보.
고성능 연산 최적화
`useMemo`를 활용하여 데이터 변경 시에만 재연산하도록 구성, 대규모 주문 내역에서도 부드러운 성능 유지.
`getKSTDateString` 유틸을 통해 타임존 차이로 인한 통계 오차를 방지하고 한국 표준시 기준의 일별 합계를 산출함.
