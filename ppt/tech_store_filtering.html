// 2. [SearchShop] URL 파라미터 감지 및 자동 필터링
const [searchParams] = useSearchParams();

useEffect(() => {

  const query = searchParams.get("search");

  if (query) {

    // 검색어 상태 동기화 및 데이터 필터링 실행

    setSearchQuery(query);

    setFilteredShops(
      MOCK_SHOPS.filter(s => s.name.includes(query))
    );

  }

}, [searchParams]);
Query Parameter Deep Linking
메인 페이지에서 매장 카드 클릭 시, 해당 매장 명칭을 search 쿼리 파라미터에 담아 전송하여 상태를 공유함.
State Synchronization
useSearchParams 훅을 통해 URL 변화를 감지하고 검색창 값과 필터링 상태를 자동으로 동기화.
Seamless Map Interaction
필터링된 매장이 하나일 경우 카카오 맵 API와 연동하여 해당 위치로 자동 이동.
페이지 간 데이터 전달을 URL 기반으로 설계하여 새로고침이나 링크 공유 시에도 동일한 검색 결과를 유지하도록 구현.
