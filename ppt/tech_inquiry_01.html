WebEditor.tsx 
// 1. 커스텀 폰트 사이즈 확장 기능 구현
const FontSize = Extension.create({
  name: 'fontSize',
  addGlobalAttributes() {
    return [{
      types: ['textStyle'],
      attributes: {
        fontSize: {
          renderHTML: attr => attr.fontSize ? 
            { style: `font-size: ${attr.fontSize}px` } : {}
        },
      },
    }]
  }
});

// 2. 에디터 코어 인스턴스 초기화
const editor = useEditor({
  extensions: [
    StarterKit, Placeholder, Image, Link,
    TextAlign, Highlight, TextStyle, Color, FontSize
  ],
  onUpdate: ({ editor }) => onChange(editor.getHTML())
});
Headless 구조 채택
UI가 고정된 에디터가 아닌 로직만 제공하는 Tiptap을 사용하여, 나다커피의 디자인 시스템(Tailwind CSS)에 완벽히 부합하는 커스텀 UI 구축.
모듈형 확장 시스템
기본 기능 외에 이미지 업로드, 폰트 크기, 컬러 선택기 등 프로젝트 전용 익스텐션을 결합하여 강력한 편집 기능 제공.
이미지 삽입 시 Base64가 아닌 클라우드 업로드 API를 연동하여 트래픽 최적화 및 에디터 성능 확보.
