Interactive Swiper Slider

MainSection1.tsx 
import React, { useState } from 'react';
import { Swiper, SwiperSlide } from 'swiper/react';
import { Navigation, Autoplay } from 'swiper/modules';
import 'swiper/css';
import 'swiper/css/navigation';

const MainSection1 = () => {
  // 1. 외부 제어 버튼을 위한 State Ref
  const [prevEl, setPrevEl] = useState<HTMLElement | null>(null);
  const [nextEl, setNextEl] = useState<HTMLElement | null>(null);

  const slides = [
    { id: 1, img: "/img/slide1.webp", title: "Signature Coffee" },
    { id: 2, img: "/img/slide2.webp", title: "Fresh Beans" }
  ];

  return (
    <div className="relative w-full h-screen group">
      
      {/* 2. Swiper Core */}
      <Swiper
        modules={[Navigation, Autoplay]}
        navigation={{ prevEl, nextEl }}
        autoplay={{ delay: 6000, disableOnInteraction: false }}
        loop={true}
        speed={800}
        className="w-full h-full"
      >
        {slides.map((slide) => (
          <SwiperSlide key={slide.id}>
            <div className="w-full h-full bg-cover bg-center"
                 style={{ backgroundImage: `url(${slide.img})` }}>
              <div className="absolute inset-0 bg-black/40 flex items-center">
                <h2 className="text-white text-5xl font-bold ml-20">{slide.title}</h2>
              </div>
            </div>
          </SwiperSlide>
        ))}
      </Swiper>

      {/* 3. Custom Navigation Buttons */}
      <div className="absolute bottom-10 right-10 z-10 flex gap-4">
        <button 
          ref={(node) => setPrevEl(node)}
          className="w-12 h-12 rounded-full border border-white/50 text-white hover:bg-white hover:text-black transition-all"
        >
          <i className="fas fa-chevron-left" />
        </button>
        <button 
          ref={(node) => setNextEl(node)}
          className="w-12 h-12 rounded-full border border-white/50 text-white hover:bg-white hover:text-black transition-all"
        >
          <i className="fas fa-chevron-right" />
        </button>
      </div>

    </div>
  );
};

export default MainSection1;
커스텀 컨트롤 바인딩
React ref와 State를 활용하여 Swiper의 기본 버튼 대신 브랜드 스타일의 커스텀 버튼을 연결.
Seamless Loop
800ms 전환 속도와 6초 Autoplay를 결합하여 끊김 없는 슬라이드 경험 제공.
스타일 재정의
라이브러리 기본 UI를 서비스 디자인에 맞게 CSS로 세밀하게 재정의.
Tailwind CSS와 결합하여 Hover Blur, Scale 등 마이크로 인터랙션 구현.
